Documentation 
// Full-Height Captures:

http://api.screenshotlayer.con/api/capture

    ? access_key = YOUR_ACCESS_KEY
    & url = http://facebook.com
    & viewport = 2560x1440   
    & fullpage = 1
    
// API returns a crisp full-height screenshot 
// of facebook.com, in iMac 27" resolution

// Thumbnails:

http://api.screenshotlayer.con/api/capture

    ? access_key = YOUR_ACCESS_KEY
    & url = http://amazon.com
    & viewport = 1440x900   
    & width = 300
    
// API returns compact 300px thumbnail of 
// amazon.com, in standard MacBook resolution

// HTTP Headers:

http://api.screenshotlayer.con/api/capture
    ? access_key = YOUR_ACCESS_KEY
    & url = http://tumblr.com
    & viewport = 375x667
    & user_agent = Mozilla/5.0 (iPhone; CPU iPhone OS 8_0_2 like Mac OS X) AppleWebKit/600.1.4 (KHTML, like Gecko) Version/8.0 Mobile/12A366 Safari/600.1.4   
    & accept_lang = es-ES
    
// API returns authentic iPhone 6 version of  
// tumblr.com, in Spanish language
